插入排序

第一次比较前两个数，小的放在前面，大的在后面，然后得到一个2个数有序列，然后把后面的一个数与该

数列比较，插入合适的位置，得到三个数的有序列，依次循环至最后一个数，就保证有序了。

1.总共需要比较length-1趟 (例如数组{1,2,3}比较两趟，第一趟得有序列{1,2}，第二趟得到有序列{1,2,3})。

2.每次把后面一个数和前面的序列每一个数比较，小于就移动，大于就停下。

  1. //插入排序
  2. void InsertSort(int arr[], int n) {
  3.     int temp, i, j;
  4.     for (i = 1; i < n; ++i) {
  5.         temp = arr[i];
  6.         j = i - 1;
  7.         while (j >= 0 && temp < arr[j]) {
  8.             arr[j + 1] = arr[j];
  9.             --j;
  10.         }
  11.         arr[j + 1] = temp;
  12.     }
  13. }
  1. //插入排序(折半插入)
  2. void BInsertSort(int A[], int n) {
  3.     int i, j, low, high, mid;
  4.     for (i = 2; i <= n; i++) {
  5.         A[0] = A[i];
  6.         low = 1;
  7.         high = i - 1;
  8.         while (low <= high) {
  9.             mid = (low + high) / 2;
  10.             if (A[mid] > A[0])
  11.                 high = mid - 1;
  12.             else
  13.                 low = mid + 1;
  14.         }
  15.         for (j = i - 1; j >= high + 1; --j) {
  16.             A[j + 1] = A[j];
  17.         }
  18.         A[high + 1] = A[0];
  19.     }
  20. }

选择排序

把第一个元素和后面所有的元素进行比较，一趟结束，最小的值出现在第一个，然后继续比较其他的数。

对于一个 长度为length的数组：

1.总共需要比较length-1趟 (例如数组{1,2,3}需要比较两趟，第一趟排除1，第二趟排除2)。

2.根据每一趟可以得到当前最小值所在位置n，然后依次和(n+1)至(length-1)比较

  1. //选择排序
  2. void SelectSort(int A[], int n) {
  3.     for (int i = 0; i < n - 1; ++i) {
  4.         int min = i;
  5.         for (int j = i + 1; j < n; ++j) {
  6.             if (A[j] < A[min])
  7.                 min = j;
  8.         }
  9.         if (min != i) Swap(A[i], A[min]);
  10.     }
  11. }
  1. //选择排序（链表）
  2. void SelectSort(LinkList &L) {
  3.     LNode *h = L, *p, *q, *r, *s;
  4.     L = NULL;
  5.     while (h != NULL) {
  6.         p = s = h;
  7.         q = r = NULL;
  8.         while (p != NULL) {
  9.             if (p->data > s->data) {
  10.                 s = p;
  11.                 r = q;
  12.             }
  13.             q = p;
  14.             p = p->next;
  15.         }
  16.         if (s == h)
  17.             h = h->next;
  18.         else
  19.             r->next = s->next;
  20.         s->next = L;
  21.         L = s;
  22.     }
  23. }

希尔排序

希尔排序是把记录按下标的一定增量分组，对每组使用直接插入排序算法排序；随着增量逐渐减少，每组包含的关键词越来越多，当增量减至1时，整个文件恰被分成一组，算法便终止。

  1. //希尔排序
  2. void ShellSort(int arr[], int n) {
  3.     int temp;
  4.     for (int gap = n / 2; gap > 0; gap /= 2) {
  5.         for (int i = gap; i < n; ++i) {
  6.             temp = arr[i];
  7.             int j;
  8.             for (j = i; j >= gap && arr[j - gap] > temp; j -= gap) {
  9.                 arr[j] = arr[j - gap];
  10.             }
  11.             arr[j] = temp;
  12.         }
  13.     }
  14. }

冒泡排序

冒泡排序的原理是依次比较两个相邻的数，小的放在前面，大的放在后面。

每一趟比较，会把最大的数放到最后面，然后排除，继续比较其他的数。

对于一个 长度为length的数组：

1.总共需要比较length-1趟 (例如数组{1,2,3}需要比较两趟，第一趟排除3，第二趟排除2)。

2.每一趟比较的次数为length - 1 - 当前趟数 i，因为每一趟会排除一个最大值，下次就不需要比较。

  1. //冒泡排序
  2. void BubbleSort(int A[], int n) {
  3.     for (int i = 0; i < n - 1; ++i) {
  4.         bool flag = false;
  5.         for (int j = n - 1; j > i; --j) {
  6.             if (A[j - 1] > A[j]) {
  7.                 int temp = A[j];
  8.                 A[j] = A[j - 1];
  9.                 A[j - 1] = temp;
  10.                 flag = true;
  11.             }
  12.         }
  13.         if (flag == false)
  14.             return;
  15.     }
  16. }
  1. //双向冒泡算法
  2. void CycleBubbleSort(int A[], int n) {
  3.     int low = 0, high = n - 1;
  4.     bool flag = true;
  5.     while (low < high && flag) {
  6.         flag = false;
  7.         int i;
  8.         for (i = low; i < high; ++i) {
  9.             if (A[i] > A[i + 1]) {
  10.                 int temp = A[i];
  11.                 A[i] = A[i - 1];
  12.                 A[i - 1] = temp;
  13.             }
  14.         }
  15.         high--;
  16.         for (i = high; i > low; i--) {
  17.             if (A[i] < A[i - 1]) {
  18.                 int temp = A[i];
  19.                 A[i] = A[i - 1];
  20.                 A[i - 1] = temp;
  21.                 flag = true;
  22.             }
  23.         }
  24.         low++;
  25.     }
  26. }

快速排序

快速排序是交换排序的一种，实质上是对冒泡排序的一种改进。在n个记录中取某一个记录的键值为标准，通常取第一个记录键值为基准，通过一趟排序将待排的记录分为小于或等于这个键值和大于这个键值的两个独立的部分，这时一部分的记录键值均比另一部分记录的键值小，然后，对这两部分记录继续分别进行快速排序，以达到这个记录的有序。

  1. //快速排序
  2. void QuickSort(int arr[], int low, int high) {
  3.     int temp;
  4.     int i = low, j = high;
  5.     if (low < high) {
  6.         temp = arr[low];
  7.         while (i < j) {
  8.             while (j > i && arr[j] >= temp)
  9.                 --j;
  10.             if (i < j) {
  11.                 arr[i] = arr[j];
  12.                 ++i;
  13.             }
  14.             while (i < j && arr[i] < temp)
  15.                 ++i;
  16.             if (i < j) {
  17.                 arr[j] = arr[i];
  18.                 ++j;
  19.             }
  20.         }
  21.         arr[i] = temp;
  22.         QuickSort(arr, low, i - 1);
  23.         QuickSort(arr, i + 1, high);
  24.     }
  25. }

堆排序

堆排序是指利用heap这种数据结构所设计的一种排序算法，是选择排序的一种。可以利用数组的特点快速定位指定索引的元素，是完全二叉树。

  1. //重建堆
  2. void sift(int arr[], int low, int high) {
  3.     int i = low, j = 2 * i + 1;
  4.     int temp = arr[i];
  5.     while (j <= high) {
  6.         if (j < high && arr[j] < arr[j + 1])
  7.             ++j;
  8.         if (temp < arr[j]) {
  9.             arr[i] = arr[j];
  10.             i = j;
  11.             j = 2 * i + 1;
  12.         } else
  13.             break;
  14.     }
  15.     arr[i] = temp;
  16. }
  1. //堆排序
  2. void HeapSort(int arr[], int n) {
  3.     int i;
  4.     int temp;
  5.     for (i = n / 2 - 1; i >= 0; --i) {
  6.         sift(arr, i, n - 1);
  7.     }
  8.     for (i = n - 1; i > 0; --i) {
  9.         temp = arr[0];
  10.         arr[0] = arr[i];
  11.         arr[i] = temp;
  12.         sift(arr, 0, i - 1);
  13.     }
  14. }

归并排序

归并排序采用了分治策略（divide-and-conquer），就是将原问题分解为一些规模较小的相似子问题，然后递归解决这些子问题，最后合并其结果作为原问题的解。

  1. //归并排序
  2. void Merge(int arr[], int low, int mid, int high) {
  3.     int i, j, k;
  4.     int n1 = mid - low + 1;
  5.     int n2 = high - mid;
  6.     int L[n1], R[n2];
  7.     for (i = 0; i < n1; i++) {
  8.         L[i] = arr[low + i];
  9.     }
  10.     for (j = 0; j < n2; j++) {
  11.         R[j] = arr[mid + 1 + j];
  12.     }
  13.     i = 0;
  14.     j = 0;
  15.     k = low;
  16.     while (i < n1 && j < n2) {
  17.         if (L[i] < R[j])
  18.             arr[k] = L[i++];
  19.         else
  20.             arr[k] = R[j++];
  21.         k++;
  22.     }
  23.     while (i < n1)
  24.         arr[k++] = L[i++];
  25.     while (j < n2)
  26.         arr[k++] = R[j++];
  27. }
  1. //归并排序
  2. void MergeSort(int arr[], int low, int high) {
  3.     if (low < high) {
  4.         int mid = (low + high) / 2;
  5.         MergeSort(arr, low, mid);
  6.         MergeSort(arr, mid + 1, high);
  7.         Merge(arr, low, mid, high);
  8.     }
  9. }
